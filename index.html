async function searchHaveIBeenPwned(email) { try { const response = await fetch(`https://haveibeenpwned.com/api/v3/breachedaccount?account=${encodeURIComponent(email)}`, { headers: { 'User-Agent': 'OSINT-Search-Engine' } }); if (response.status === 200) { const breaches = await response.json(); return { found: true, count: breaches.length, data: breaches, source: 'HaveIBeenPwned' }; } return { found: false, message: 'לא נמצא בדליפות', source: 'HaveIBeenPwned' }; } catch(e) { return { error: e.message }; } } async function searchGitHub(username) { try { const response = await fetch(`https://api.github.com/search/users?q=${encodeURIComponent(username)}`); const data = await response.json(); if (data.items && data.items.length > 0) { return { found: true, count: data.items.length, data: data.items.slice(0, 5), source: 'GitHub' }; } return { found: false, source: 'GitHub' }; } catch(e) { return { error: e.message }; } } function buildSocialSearches(term) { return [ { title: 'Google', icon: '🔍', url: `https://www.google.com/search?q=${encodeURIComponent(term)}`, description: 'חפש בGoogle' }, { title: 'Twitter', icon: '𝕏', url: `https://twitter.com/search?q=${encodeURIComponent(term)}`, description: 'חפש בTwitter' }, { title: 'Reddit', icon: '🔴', url: `https://www.reddit.com/search/?q=${encodeURIComponent(term)}`, description: 'חפש בReddit' }, { title: 'LinkedIn', icon: '💼', url: `https://www.linkedin.com/search/results/people/?keywords=${encodeURIComponent(term)}`, description: 'חפש בLinkedIn' }, { title: 'Facebook', icon: '📘', url: `https://www.facebook.com/search/top/?q=${encodeURIComponent(term)}`, description: 'חפש בFacebook' }, { title: 'Instagram', icon: '📷', url: `https://www.instagram.com/explore/tags/${encodeURIComponent(term)}/`, description: 'חפש בInstagram' }, { title: 'YouTube', icon: '🎥', url: `https://www.youtube.com/results?search_query=${encodeURIComponent(term)}`, description: 'חפש בYouTube' }, { title: 'Stack Overflow', icon: '⚡', url: `https://stackoverflow.com/search?q=${encodeURIComponent(term)}`, description: 'חפש בStack Overflow' } ]; } async function performSearch() { const searchInput = document.getElementById('searchInput').value.trim(); const searchType = document.getElementById('searchType').value; if (!searchInput) { showError('❌ אנא הכנס ערך לחיפוש'); return; } document.getElementById('loading').style.display = 'block'; document.getElementById('results').style.display = 'none'; document.getElementById('error').style.display = 'none'; try { let results = []; if (searchType === 'email') { const breach = await searchHaveIBeenPwned(searchInput); results.push({ category: '🔐 דליפות מידע', ...breach }); results.push({ category: '📱 רשתות חברתיות', sources: buildSocialSearches(searchInput) }); } else if (searchType === 'username') { const github = await searchGitHub(searchInput); results.push({ category: '👨‍💻 GitHub', ...github }); results.push({ category: '📱 רשתות חברתיות', sources: buildSocialSearches(searchInput) }); } else if (searchType === 'phone' || searchType === 'name') { results.push({ category: '📱 חיפוש', sources: buildSocialSearches(searchInput) }); } document.getElementById('loading').style.display = 'none'; displayResults(results); showSuccess('✅ חיפוש הושלם!'); } catch(e) { document.getElementById('loading').style.display = 'none'; showError('❌ שגיאה: ' + e.message); } } function displayResults(results) { const resultsList = document.getElementById('resultsList'); resultsList.innerHTML = ''; results.forEach(category => { if (category.sources) { let html = `

${category.category}

`; category.sources.forEach(source => { html += `
${source.icon}
${source.title}
${source.description}
`; }); html += '
'; resultsList.innerHTML += html; } else if (category.found) { let html = `

${category.category}

✅ נמצא!

${category.source}: ${category.count} תוצאות

`; if (category.data) { category.data.forEach(item => { if (item.profile_url) html += `

👤 ${item.login}

`; else if (item.Title) html += `

📋 ${item.Title} - ${item.BreachDate}

`; }); } html += '
'; resultsList.innerHTML += html; } }); document.getElementById('results').style.display = 'block'; } function showError(message) { const errorDiv = document.getElementById('error'); errorDiv.textContent = message; errorDiv.style.display = 'block'; } function showSuccess(message) { const successDiv = document.getElementById('success'); successDiv.textContent = message; successDiv.style.display = 'block'; setTimeout(() => { successDiv.style.display = 'none'; }, 3000); } document.addEventListener('DOMContentLoaded', () => { document.getElementById('searchInput').addEventListener('keypress', (e) => { if (e.key === 'Enter') performSearch(); }); });
